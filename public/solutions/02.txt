menu = []

class BakeryItem:
    def __init__(this, name, price, category):
        this.name = name
        this.price = price
        this.category = category

    def print_details(this):
        print("Item: " + this.name + " | Category: " + this.category + " | Price: $" + str(this.price))


item1 = BakeryItem("Chocolate Cake", 15.00, "Cakes")
item2 = BakeryItem("Chocolate Chip Cookie", 2.50, "Cookies")
item3 = BakeryItem("Vanilla Fudge", 3.00, "Sweets")

menu.append(item1)
menu.append(item2)
menu.append(item3)

for item in menu:
    item.print_details()
